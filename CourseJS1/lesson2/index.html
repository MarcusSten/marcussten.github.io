6. Реализовать функцию с тремя параметрами: function mathOperation(arg1, arg2, operation), где arg1, arg2 — значения аргументов, operation — строка с названием операции. В зависимости от переданного значения выполнить одну из арифметических операций (использовать функции из пункта 5) и вернуть полученное значение (применить switch).

var x = parseInt(Math.random() * 10);
var y = parseInt(Math.random() * 10);
var operation = prompt("Даны числа x = " + x + " и y = " + y + ". Укажите каое действие с числами выполнить (используйте кнопки: + - * / )");

function mathOperation(operation, x, y){

    function addition(x,y){
        return x + y;
    }
    function subtraction(x,y){
        return x - y;
    }
    function multiplication(x,y){
        return x * y;
    }
    function division(x,y){
        return x / y;
    }

    switch(operation) 
    {
        case '+' :
            return addition(x,y);
        
        case '-' :
            return subtraction(x,y);
        
        case '*' :
            return multiplication(x,y);

        case '/' :
            return division(x,y);  
        
        default :
            return("Вы ввели неверный символ.");
    }
}

alert("Итого " + x + operation + y + "=" + mathOperation(operation, x, y));

7. * Сравнить null и 0. Объяснить результат.

alert(null == 0);
alert(null === 0);

Оператор == использует так называемый абстрактный алгоритм сравнения для равенств, возвращая в результате true или false. Следуя этому алгоритму приходим к выводу, что эти типы отличаются. При этом 0 не является undefined. Поэтому возвращается false.
Так же проверяются следуюшщие утверждения (используя абстрактный алгоритм сравнения для отношений): 

null > 0;   // false
null == 0;  // false
null >= 0;  // true
null > 0;   // false
null < 0;   // false

8. * С помощью рекурсии организовать функцию возведения числа в степень. Формат: function power(val, pow), где val — заданное число, pow –— степень.

var val = parseInt(prompt("Задайте произвольное число"));
var pow = parseInt(prompt("Задайте степень числа"));

function power(val, pow){
    if(pow == 1){
        return val;
    } else{
    return val * power(val, pow - 1);
    }
}
alert(power(val, pow));
