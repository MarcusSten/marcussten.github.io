2. Не выполняя кода, ответить, что выведет браузер и почему: 
if (!("a" in window)) {
    var a = 1;
}
alert(a);
var b = function a(x) {
    x && a(--x);
};
alert(a);
function a(x) {
    return x * 2;
}
var a;
alert(a);
function b(x, y, a) {
    arguments[2] = 10;
    alert(a);
}
b(1, 2, 3);
function a() {
    alert(this);
}
a.call(null);

if (!("a" in window)) {
    var a = 1;
}
alert(a);
Ответ: Зависит от того как запускается код в локальной или глобальной зоне видимости. Если первое, то а=1, т.к. никакой глобальной "а" нет, а если второе, то undefined.
----------------------
var b = function a(x) {
    x && a(--x);
};
alert(a);
Ответ: undefined. Вижу примерно так: b = function a(){ ...по имени «a» функцию видно только здесь}.
----------------------
function a(x) {
    return x * 2;
}
var a;
alert(a);
Ответ: Наверное выведется сама функция, раз аргумент не передан. А вот если передать, то будет результат умножения.
----------------------
function b(x, y, a) {
    arguments[2] = 10;
    alert(a);
}
b(1, 2, 3);
Ответ: 10. Присваивание внутри функции переопределит "а".

function a() {
