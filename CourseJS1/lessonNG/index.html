6. Написать функцию, которой передаем имя, и она возраващает приветствие в зависимости от времени суток (Доброе утро\день\вечер\ночи Иван)

Смотреть ответ 

function getDay(){
    let name = prompt("Укажите свое имя");
    let d = new Date();
    let hours = d.getHours();
    
    if (hours > 6 && hours < 12){
        alert("Доброе утро " + name);
    } else if (hours >= 12 && hours < 18){
        alert("Доброе день " + name);
    } else if (hours >= 18 && hours <= 23){
        alert("Доброе вечер " + name);
    } else {
        alert("Доброй ночи " + name);
    }
}
getDay();

    {name: 'Oleg', age: 35, salary: 700},
    {name: 'Givi', age: 23, salary: 1000},
    {name: 'Igor', age: 50, salary: 1100},
    {name: 'Pavel', age: 30, salary: 300}

16. Создайте функцию multiplyNumeric(obj), которая умножает все числовые свойства объекта obj на 2.

let obj = {name: 'Oleg', age: 35, salary: 700};
function multiplyNumeric(obj){
    for(let key in obj){
        if(typeof obj[key] == 'number'){
            obj[key] *= 2;
        };
    }
}
